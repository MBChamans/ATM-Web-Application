CREATE TABLE ICICI (
cardno BIGINT(16) UNSIGNED PRIMARY KEY,
pin INT(4) NOT NULL,
accountno BIGINT(12) UNSIGNED NOT NULL,
name VARCHAR(30) NOT NULL,
balance INT(10) NOT NULL
);
CREATE TABLE HDFC(
cardno BIGINT(16) UNSIGNED PRIMARY KEY,
pin INT(4) NOT NULL,
accountno BIGINT(12) UNSIGNED NOT NULL,
name VARCHAR(30) NOT NULL,
balance INT(10) NOT NULL
);
CREATE TABLE SBI(
cardno BIGINT(16) UNSIGNED PRIMARY KEY,
pin INT(4) NOT NULL,
accountno BIGINT(12) UNSIGNED NOT NULL,
name VARCHAR(30) NOT NULL,
balance INT(10) NOT NULL
)
